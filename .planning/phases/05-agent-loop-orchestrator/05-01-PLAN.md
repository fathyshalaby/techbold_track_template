    - parse with missing benefitCheck throws ZodError
  </behavior>
  <action>
    Create backend/src/ai/types.ts. Import z from 'zod'.

      command: z.string()
      purpose: z.string()
      expectedSignal: z.string()
      riskNotes: z.string()
      isReadOnly: z.boolean()

      rationale: z.string()
      rollbackCommand: z.string()

      persistenceCheck: z.string()
      evidence: z.array(z.string())
